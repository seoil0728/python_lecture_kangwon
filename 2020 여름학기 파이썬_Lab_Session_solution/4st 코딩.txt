# 입력된 세 숫자를 비교하기 위한 함수 작성, 각 크기를 비교하며 가장 큰 숫자를 출력
def compare(a, b, c):
    if a > b:
        if a > c:
            return a
        else:
            return c
    else:
        if b > c:
            return b
        return c


# main
# 정수형의 세 숫자를 입력받음
a = int(input("Enter a: "))
b = int(input("Enter b: "))
c = int(input("Enter c: "))

#가장 큰 수를 max에 저장
max = compare(a, b, c)
#출력
print(a, b, c, "중 가장 큰 값은", max, "입니다.")

    #다음의 조건에 따른 id, passwd를 검사하고 해당하는 오류문을 출력한다.
